        <script>
           function solution(a){  
                let answer= a;
                let sum= answer.reduce((a,b)=>a+b,0);
                for( var i= 0 ; i<answer.length-1; i++){
                    for(var j=i+1; j<answer.length; j++){
                        if((sum-(answer[i]+answer[j]))==100){
                            answer.splice(i,1);
                            answer.splice(j,1);
                        }
                    }
                }
              
                return answer;
            }
          
            
            
            let a=[20, 7, 23, 19, 10, 15, 25, 8, 13];
            console.log(solution(a));
        </script>
