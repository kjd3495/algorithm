        <script>
            function solution(a){
                let answer=[];
                let cnt=1;
                for(var i=0; i<a.length; i++){
                    for(var j=0; j<a.length; j++){
                        if(a[i]<a[j]){
                            cnt++
                        }
                    }
                    answer.push(cnt);
                    cnt=1;
                    }
                
                return answer;
            }
            
            
            let a = [92,92,92,100,76];
            console.log(solution(a));
        </script>
